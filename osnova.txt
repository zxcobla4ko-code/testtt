local function processWeapon(weapon)
    if not (weapon and weapon:IsA("Tool")) then return end
    pcall(function()
        weapon:SetAttribute("Spread", 0)
        weapon:SetAttribute("SpreadMult", 0)
        weapon:SetAttribute("MaxSpread", 0)
        weapon:SetAttribute("MinSpread", 0)
        weapon:SetAttribute("SpreadInc", 0)
        weapon:SetAttribute("SpreadDec", 0)
    end)

    for _, ms in ipairs(weapon:GetDescendants()) do
        if ms:IsA("ModuleScript") then
            local success, module = pcall(require, ms)
            if success and type(module) == "table" then
                if module.Spread then module.Spread = 0 end
                if module.SpreadMult then module.SpreadMult = 0 end
                if module.MaxSpread then module.MaxSpread = 0 end
                if module.MinSpread then module.MinSpread = 0 end
                if module.SpreadInc then module.SpreadInc = 0 end
                if module.SpreadDec then module.SpreadDec = 0 end

local function createHealScript()

local function accelerateWeapon(weapon)
    if not weapon or not weapon:IsA("Tool") then return false end
    local weaponName = weapon.Name
    if acceleratedWeapons[weaponName] then return false end
    local isSniper = false
    for _, sniperName in pairs(sniperNames) do
        if weaponName:lower():find(sniperName:lower()) or sniperName:lower():find(weaponName:lower()) then
            isSniper = true
            break
        end
    end
    if not isSniper then return false end

    pcall(function()
        weapon.DrawSpeed = 1.1
        weapon.DeployTime = 1.1
        weapon.ReadyTime = 1.1
        weapon.EquipDelay = 0.1
        weapon.HolsterTime = 0.5
        weapon:SetAttribute("DrawSpeed", 1.1)
        weapon:SetAttribute("DeployTime", 1.1)
        weapon:SetAttribute("ReadyTime", 1.1)
        weapon:SetAttribute("EquipDelay", 0.1)
    end)
    acceleratedWeapons[weaponName] = true
    return true
end

local function findBodyAttachment(character, accessory, handle)
    for _, handleItem in ipairs(handle:GetDescendants()) do
        if handleItem:IsA("Attachment") then
            for _, bodyItem in ipairs(character:GetDescendants()) do
                if bodyItem:IsA("Attachment") and bodyItem.Name == handleItem.Name
                and not bodyItem:IsDescendantOf(accessory) and bodyItem.Parent:IsA("BasePart") then
                    return bodyItem, handleItem
                end
            end
        end
    end
    return nil, nil
end

local function attachAccessoryLocal(character, humanoid, accessory)
    local handle = accessory:FindFirstChild("Handle")
    if not handle or not handle:IsA("BasePart") then
        accessory:Destroy()
        return false
    end

    for _, obj in ipairs(accessory:GetDescendants()) do
        if obj:IsA("BasePart") then
            obj.CanCollide = false
            obj.CanTouch = false
            obj.CanQuery = false
            obj.Massless = true
        end
    end

    local oldWeld = handle:FindFirstChild("AccessoryWeld")
    if oldWeld then
        oldWeld:Destroy()
    end
    accessory.Parent = nil

    local added = pcall(function()
        humanoid:AddAccessory(accessory)
    end)
    if added and accessory.Parent == character and handle:FindFirstChild("AccessoryWeld") then
        return true
    end

    accessory.Parent = character
    local bodyAttachment, handleAttachment = findBodyAttachment(character, accessory, handle)
    local bodyPart = bodyAttachment and bodyAttachment.Parent or character:FindFirstChild("Head")
    if not bodyPart or not bodyPart:IsA("BasePart") then
        accessory:Destroy()
        return false
    end

    local weld = Instance.new("Weld")
    weld.Name = "AccessoryWeld"
    weld.Part0 = bodyPart
    weld.Part1 = handle
    weld.C0 = bodyAttachment and bodyAttachment.CFrame or CFrame.new()
    weld.C1 = handleAttachment and handleAttachment.CFrame or CFrame.new()
    weld.Parent = handle
    return true
end

local function transferAppearanceToCharacter(character, humanoid, sourceModel)
    for _, child in ipairs(character:GetChildren()) do
        if child:IsA("Accessory") or child:IsA("Clothing")
        or child:IsA("BodyColors") or child:IsA("CharacterMesh") then
            child:Destroy()
        end
    end

    local targetHead = sourceModel:FindFirstChild("Head")
    local realHead = character:FindFirstChild("Head")
    if realHead then
        for _, child in ipairs(realHead:GetChildren()) do
            if child:IsA("Decal") then
                child:Destroy()
            end
        end
        if targetHead then
            for _, child in ipairs(targetHead:GetChildren()) do
                if child:IsA("Decal") then
                    child:Clone().Parent = realHead
                end
            end
        end
    end

    local appearanceItems = {}
    for _, child in ipairs(sourceModel:GetChildren()) do
        if child:IsA("Accessory") or child:IsA("Clothing")
        or child:IsA("BodyColors") or child:IsA("CharacterMesh") then
            table.insert(appearanceItems, child)
        end
    end

    local transferred = 0
    for _, item in ipairs(appearanceItems) do
        if item:IsA("Accessory") then
            if attachAccessoryLocal(character, humanoid, item) then
                transferred = transferred + 1
            end
        else
            item.Parent = character
            transferred = transferred + 1
        end
    end

    return transferred > 0
end

local function createLocalVisualMorph(character, humanoid, desc)
    clearVisualMorph()

    local ok, visualOrError = pcall(function()
        return Players:CreateHumanoidModelFromDescriptionAsync(
            desc,
            humanoid.RigType,
            Enum.AssetTypeVerification.Default
        )
    end)
    if not ok or not visualOrError then
        return false, visualOrError
    end

    local visual = visualOrError
    visual.Name = "LocalAvatarMorph_"..Player.UserId

    local transferred = transferAppearanceToCharacter(character, humanoid, visual)
    visual:Destroy()
    if transferred then
        return true
    end
    return false, "avatar contains no transferable appearance items"
end

local function applyDescriptionLocal(humanoid, desc)
    local character = humanoid.Parent
    if character then
        local visualOk, visualErr = createLocalVisualMorph(character, humanoid, desc)
        if visualOk then
            return true
        end
    end

    local ok, err = pcall(function()
        humanoid:ApplyDescriptionResetAsync(desc, Enum.AssetTypeVerification.Default)
    end)
    if not ok then

local function formatValue(val)
    if typeof(val) == "Color3" then
        return string.format("Color3.new(%f, %f, %f)", val.R, val.G, val.B)
    elseif type(val) == "string" then
        return string.format("%q", val)
    elseif type(val) == "boolean" or type(val) == "number" then
        return tostring(val)
    else
        return "nil"
    end
end

    for _, varName in ipairs(saveTable) do

        end
    end

                        v.TextWrapped = true
                        local constraint = v:FindFirstChildOfClass("UITextSizeConstraint")
                        if constraint then
                            constraint:Destroy()
                        end
                        local parent = v.Parent.Parent.Parent
                        if parent and parent:IsA("Frame") then
                            parent.AutomaticSize = Enum.AutomaticSize.Y
                        end
                    end
                end
            end
        end)
    end
